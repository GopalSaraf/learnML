 10    @staticmethod
 11    def confusion_matrix(y_pred: np.ndarray, y_test: np.ndarray) -> np.ndarray:
 12        """
 13        Returns the confusion matrix of the model.
 14
 15        Parameters
 16        ----------
 17        y_pred : np.ndarray
 18            The model prediction array of shape (n_samples, ) or (1, n_samples)
 19        y_test : np.ndarray
 20            The testing array of shape (n_samples, ) or (1, n_samples)
 21
 22        Returns
 23        -------
 24        np.ndarray
 25            The confusion matrix of the model
 26        """
 27        if len(y_pred.shape) == 2:
 28            y_pred = y_pred.reshape(-1)
 29
 30        if len(y_test.shape) == 2:
 31            y_test = y_test.reshape(-1)
 32
 33        assert y_pred.shape == y_test.shape, (
 34            f"y_pred and y_test should contain same number of samples. "
 35            f"y_pred contains {y_pred.shape[0]} whereas y_test contain {y_test.shape[0]} samples"
 36        )
 37
 38        unique = np.unique(y_test)
 39        n = len(unique)
 40
 41        matrix = np.zeros((n, n), dtype=int)
 42
 43        for i in range(len(y_test)):
 44            matrix[y_test[i], y_pred[i]] += 1
 45
 46        return matrix
